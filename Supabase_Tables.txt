-- Supabase SQL Schema for ShEC CSE App

-- 1. Notices Table
CREATE TABLE notices (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    category TEXT NOT NULL CHECK (category IN ('club', 'department')),
    title TEXT NOT NULL,
    subtitle TEXT NOT NULL,
    description TEXT DEFAULT '',
    image_path TEXT DEFAULT '',
    tags TEXT[] NOT NULL DEFAULT '{}',
    tag_color BIGINT NOT NULL,
    date TEXT NOT NULL,
    is_pinned BOOLEAN NOT NULL DEFAULT FALSE,
    icon_code_point BIGINT NOT NULL,
    icon_color BIGINT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 2. Jobs Table
CREATE TABLE jobs (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    category TEXT NOT NULL CHECK (category IN ('recommended', 'recent')),
    company TEXT NOT NULL,
    role TEXT NOT NULL,
    location TEXT NOT NULL,
    salary TEXT NOT NULL,
    deadline TEXT NOT NULL,
    job_type TEXT NOT NULL,
    type_color BIGINT NOT NULL,
    icon_code_point BIGINT NOT NULL,
    icon_color BIGINT NOT NULL,
    is_starred BOOLEAN NOT NULL DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 3. Contests Table
CREATE TABLE contests (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    title TEXT NOT NULL,
    platform TEXT NOT NULL,
    level TEXT NOT NULL,
    date TEXT NOT NULL,
    url TEXT NOT NULL,
    icon_color BIGINT NOT NULL,
    is_course BOOLEAN NOT NULL DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 4. Resources Table
CREATE TABLE resources (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name TEXT NOT NULL,
    size TEXT NOT NULL,
    date TEXT NOT NULL,
    session TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Note: In Supabase SQL Editor, `uuid_generate_v4()` comes from the `uuid-ossp` extension, 
-- which is usually enabled by default. If it fails, run: CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 5. Gallery Table
CREATE TABLE gallery (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    title TEXT NOT NULL,
    subtitle TEXT NOT NULL,
    image_path TEXT NOT NULL,
    icon_code_point BIGINT NOT NULL,
    icon_color BIGINT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- ==========================================
-- ROW LEVEL SECURITY (RLS) POLICIES
-- ==========================================
-- To allow public read/write access (since we are not using authentication),
-- we must explicitly create these policies.

ALTER TABLE notices ENABLE ROW LEVEL SECURITY;
ALTER TABLE jobs ENABLE ROW LEVEL SECURITY;
ALTER TABLE contests ENABLE ROW LEVEL SECURITY;
ALTER TABLE resources ENABLE ROW LEVEL SECURITY;
ALTER TABLE gallery ENABLE ROW LEVEL SECURITY;

-- Notices Policies
CREATE POLICY "Enable public read access for notices" ON notices FOR SELECT USING (true);
CREATE POLICY "Enable public insert for notices" ON notices FOR INSERT WITH CHECK (true);
CREATE POLICY "Enable public update for notices" ON notices FOR UPDATE USING (true) WITH CHECK (true);
CREATE POLICY "Enable public delete for notices" ON notices FOR DELETE USING (true);

-- Jobs Policies
CREATE POLICY "Enable public read access for jobs" ON jobs FOR SELECT USING (true);

CREATE POLICY "Enable public update for jobs" ON jobs FOR UPDATE USING (true) WITH CHECK (true);
CREATE POLICY "Enable public delete for jobs" ON jobs FOR DELETE USING (true);

-- Contests Policies
CREATE POLICY "Enable public read access for contests" ON contests FOR SELECT USING (true);
CREATE POLICY "Enable public insert for contests" ON contests FOR INSERT WITH CHECK (true);
CREATE POLICY "Enable public update for contests" ON contests FOR UPDATE USING (true) WITH CHECK (true);
CREATE POLICY "Enable public delete for contests" ON contests FOR DELETE USING (true);

-- Resources Policies
CREATE POLICY "Enable public read access for resources" ON resources FOR SELECT USING (true);
CREATE POLICY "Enable public insert for resources" ON resources FOR INSERT WITH CHECK (true);

-- ==========================================
-- PROFILES AND AUTHENTICATION
-- ==========================================

-- Create a type for user roles
CREATE TYPE user_role AS ENUM ('member', 'committee', 'superuser');

-- 6. Profiles Table
CREATE TABLE profiles (
    id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
    first_name TEXT NOT NULL,
    last_name TEXT NOT NULL,
    class_id TEXT NOT NULL,
    batch TEXT NOT NULL,
    session TEXT NOT NULL,
    du_reg TEXT NOT NULL,
    phone TEXT NOT NULL,
    profile_pic TEXT,
    role user_role DEFAULT 'member'::user_role NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;

-- Profiles Policies
-- Anyone can read profiles
CREATE POLICY "Enable public read access for profiles" ON profiles FOR SELECT USING (true);

-- Users can insert their own profile
CREATE POLICY "Users can insert their own profile" ON profiles FOR INSERT WITH CHECK (auth.uid() = id);

-- Users can update their own profile, Superusers can update any profile
CREATE POLICY "Users can update own profile, Superusers can update all" ON profiles FOR UPDATE USING (
    auth.uid() = id OR 
    (SELECT role FROM profiles WHERE id = auth.uid()) = 'superuser'::user_role
);

-- Superusers can delete profiles
CREATE POLICY "Superusers can delete profiles" ON profiles FOR DELETE USING (
    (SELECT role FROM profiles WHERE id = auth.uid()) = 'superuser'::user_role
);

-- Note: We do not need a trigger to automatically create profiles if we are going to 
-- handle profile creation explicitly during the sign-up flow in the app. 
-- However, for robustness, we can keep the profile table strictly mapped.

-- ==========================================
-- 8. APP UPDATES SYSTEM
-- ==========================================

CREATE TABLE app_updates (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    version TEXT NOT NULL,
    build_number INT NOT NULL,
    download_url TEXT NOT NULL,
    release_notes TEXT DEFAULT '',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

ALTER TABLE app_updates ENABLE ROW LEVEL SECURITY;

-- Anyone can read/check updates
CREATE POLICY "Enable public select for app_updates" ON app_updates FOR SELECT USING (true);

-- Only Superusers can publish/insert, update, or delete updates
CREATE POLICY "Enable superuser insert for app_updates" ON app_updates FOR INSERT WITH CHECK (
    (SELECT role FROM profiles WHERE id = auth.uid()) = 'superuser'::user_role
);

CREATE POLICY "Enable superuser update for app_updates" ON app_updates FOR UPDATE USING (
    (SELECT role FROM profiles WHERE id = auth.uid()) = 'superuser'::user_role
);

CREATE POLICY "Enable superuser delete for app_updates" ON app_updates FOR DELETE USING (
    (SELECT role FROM profiles WHERE id = auth.uid()) = 'superuser'::user_role
);
